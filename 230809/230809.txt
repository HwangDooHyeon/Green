#include <stdio.h>
#include <string>

using namespace std;



int main(void)
{
	printf("%d\n", sizeof(char));
	printf("%d\n", sizeof(string));
	printf("%d\n", sizeof(short));
	printf("%d\n", sizeof(int));
	printf("%d\n", sizeof(long));
	printf("%d\n", sizeof(float));
	printf("%d\n", sizeof(double));

	/*
	char; 문자
	string; 문자열
	short; 16비트 정수
	int; 운영체제에 따라 결정
	long; 32비트 정수
	float; 소수점 7자리까지
	double; 소수점 15자리까지
	*/

	return 0;
}
